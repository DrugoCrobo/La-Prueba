    {\large \textbf{Preliminares}}
\end{center}

\begin{defi}[Conjunto Directo]
Sea $L$ una retícula.\par 
Decimos que $\left\{c_i\mid i\in I\right\}\subseteq L$ es un conjunto directo si para todo $i,j\in I$ existe $k\in I$ tal que $c_i\vee c_j\leq c_k$.
\end{defi}

\begin{defi}[Compacto]
Sea $L$ una retícula.\par 
Decimos que $fin L$ es compacto si siempre que $f\leq\bigvee S$, para algún conjunto directo $S\subseteq L$, entonces existe un $x\in S$ tal que $f\leq x$.
\end{defi}

\begin{defi}[Compactamente Generada]
Sea $L$ una retícula completa.\par 
Decimos que $L$ es compactamente generada si para todo $c\in L$ existe un conjunto $S=\left\{s\in L\mid s\text{ es compacto }\right\}\subseteq L$ y $c=\bigvee S$.
\end{defi}

\begin{defi}[Noetheriana]
Una retícula $L$ es noetheriana si satisface la condición de cadena ascendente sobre sus elementos, es decir, que dada una cadena ascendente $a_1\leq a_2\leq a_3\leq\dots$ entonces existe un $k\in \mathbb{N}$ tal que $a_k=a_{k+1}=a_{k+2}=\dots$
\end{defi}


\begin{defi}[Ideal]
Sea $L$ una retícula y $I\subseteq L$. Decimos que $I$ es un ideal si
\begin{enumerate}
    \item Para todo $x,y\in I$ se tiene que $x\vee y\in I$
    \item Sean $t\in L$ y $x\in I$ si $t\leq x$ entonces $t\in I$
\end{enumerate}
\end{defi}

Obs: Denotamos por $I\left(L\right)$ el conjunto de todos los ideales de $L$. Junto con la relación de inclusión de conjuntos, este es una retícula completa.

\begin{defi}[Ideal Principal]
Sea $L$ una retícula y $a\in L$.\par 
El conjunto de todas las cotas inferiores de $a$ es un ideal particular llamdo el ideal principal y se denotan $(a]=\sfrac{a}{0}=\left\{x\in L\mid x\leq a\right\}$.
\end{defi}

\begin{defi}[Esencial]
Sea $L$ una retícula con $0$ y $1$.\par
Un elemento $e\in L$ es esencial si para todo $x\in L$ distinto del $0$ se tiene que $e\wedge x\neq 0$
\end{defi}

Obs: El conjunto de todos los elementos esenciales de $L$ lo denotamos por $E\left(L\right)$

\begin{defi}[Zoclo]
Sea $L$ una retícula con $0$.\par
El supremo de todos los átomos en $L$ es llamado zoclo. Se denota $Zoc\left(L\right)$.
\end{defi}

\begin{defi}[Independiente]
Sean $L$ una retícula y $S\subseteq L$ finito. Decimos que $S$ es independiente si para toda $a\in L$ se tiene que $a\wedge\bigvee\left(S\setminus \left\{a\right\}\right)=0$.
\end{defi}

Obs: Un subconjunto arbitrario de $L\setminus\left\{0\right\}$ es independiente si todos subconjuntos finitos son independientes.

\begin{defi}[Modular]
Sea $L$ una retícula.\par
Decimos que $L$ es modular si para todo $a,b,c\in L$ con $b\leq a$ tenemos que $a\wedge\left(b\vee c\right)=b\vee\left(a\wedge c\right)$.
\end{defi}

\begin{defi}[Continua Superiormente]
Sea $L$ una retícula.\par
Decimos que $L$ es continua superiormente si para todo subconjunto directo $\left\{c_i\mid i\in I\right\}$ y $a\in L$ tenemos que $a\wedge\bigvee\limits_{i\in I} c_i=\bigvee\limits_{i\in I}\left(a\wedge c_i\right)$.
\end{defi}

\begin{defi}[Seudocomplemento]
Sean $L$ una retícula con $0$ y $a\in L$.\par
Decimos que $b\in L$ es un seudocomplemento de $a$ syss $b$ es máximo en el conjunto $\left\{c\in L\mid c\wedge a=0\right\}$.
\end{defi}

Obs: al conjunto de suedocomplementos lo denotamos $P\left(L\right)$.

\begin{defi}[Extensión Esencial]
Sean $L$ una retícula y $a,b\in L$.\par
Si $a\leq b$ y $a$ es esencial en $\left[0,b\right]$, entonces decimos que $b$ es una extensión esencial de $a$ en $L$. Lo denotamos como $a<_e b$.
\end{defi}

Obs: otra forma de escribir la definición anterior: $b$ es una extensión esencial de $a$ syss $a\in E\left(\sfrac{b}{0}\right)$.

\begin{defi}[Cerrado]
Sean $L$ una retícula y $c\in L$.\par
Decimos que $c$ es cerrado si no tienes extenciones esenciales propias, es decir, que el único elemento que puedo tomar mayor a él, es el propio $c$.
\end{defi}

Obs: al conjunto de los cerrados los denotamos como $C\left(L\right)$.

\begin{defi}[Uniforme]
Sean $L$ una retícula y $a\in L\setminus\left\{0\right\}$.\par
Decimos que $a$ es uniforme si para cualesquiera $x,y\in\sfrac{a}{0}$ mayores que $0$, se tiene que $0<x\wedge y$.
\end{defi}

\begin{defi}[Dimensión Uniforme o de Goldie]
Sea $L$ una retícula.\par
Decimos que $L$ tiene dimensión uniforme si existe una familia $\left\{a_1,\dots, a_n\right\}$ independiente cuyos elementos son uniformes $\bigvee\limits^n_{i-1}a_i$ es esencial en $L$.
\end{defi}

Obs: a la familia $\left\{a_1,\dots, a_n\right\}$ le llamaremos conjunto de Goldie.\par
Obs: a la dimensión uniforme de $L$ la denotamos como $u-dim\left(L\right)=n$.

\begin{defi}[Artiniana]
Sea $L$ una retícula.\par
Decimos que $L$ es artiniana si satisface la condición de cadena descendente sobre sus elementos.
\end{defi}
